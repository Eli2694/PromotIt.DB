Country nvarchar(15),City nvarchar(15),HomeAddress nvarchar(60),PhoneNumber nvarchar(15))

drop table Orders

--פוקציה בשביל להוריד מוצר אחד מהמלאי
create procedure UpdateUnitsInStockAfterPurchase @ProductID int
as
begin
 update Products 
 set UnitsInStock = UnitsInStock - 1
 where ProductID = ProductID
end

--4 פונקציה בשביל נציג עסקי שיקבל רק את ההזמנות למוצרים אותם הוא תרם
create procedure GetOrdersThatBelongToMe @Email nvarchar(255)
as
begin
    declare @UserID int
	select @UserID=(select UserID from Users where Email=@Email)
	select Orders.ProductID,Country,City,HomeAddress,PhoneNumber,ProductName,UnitPrice,UnitsInStock
	from Orders 
	inner join Products
	on Orders.ProductID = Products.ProductID
	where RepresantativeID = @UserID
end
