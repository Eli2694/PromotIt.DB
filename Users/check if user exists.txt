create procedure checkNewSiteUser @FullName varchar(50),@Email nvarchar(255)
as
begin
 if exists(select UserID from Users where FullName = @FullName and Email=@Email)
 begin
  declare @answer varchar(100)
  select @answer = 'המשתמש נמצא , אין צורך לבצע פעולה נוספת'

 end
 else
 begin
 
	insert into Users values(@FullName,@Email)
 end
end
